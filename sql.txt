WITH material_cost_per_product AS (
    SELECT
        p.id AS product_id,
        SUM(pm.quantity * m.cost) AS total_material_cost
    FROM public.product p
    JOIN public.production pr ON pr.product_id = p.id
    JOIN public.production_material pm ON pm.production_id = pr.id
    JOIN public.material m ON m.id = pm.material_id
    GROUP BY p.id
)

SELECT
    o.id AS order_id,
    c.name AS customer_name,
    SUM(oi.quantity * COALESCE(mcp.total_material_cost, 0)) AS total_order_cost
FROM public.orders o
JOIN public.customer c ON c.id = o.customer_id
JOIN public.order_item oi ON oi.order_id = o.id
LEFT JOIN material_cost_per_product mcp ON mcp.product_id = oi.product_id
WHERE o.id = 1  -- ВЫБОР ID
GROUP BY o.id, c.name;
